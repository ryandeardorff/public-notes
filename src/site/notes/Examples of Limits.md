$$
\lim_{x\rightarrow 3} x^{2}= 3^{2} = 9
$$
$$
\lim_{x\rightarrow 0} |x| = |0| = 0
$$
$$
\lim_{x\rightarrow 0} \frac{|x|}{x} = \text{Does Not Exist}
$$
$$
\lim_{x\rightarrow 0} \frac{1}{x^{2}} = \infty
$$
